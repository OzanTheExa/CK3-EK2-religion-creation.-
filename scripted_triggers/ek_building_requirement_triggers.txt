ek_rumare_tradeports_enabled = {
	trigger_if = {
		limit = {
			geographical_region = mundus_tamriel_cyrodiil_rumare_niben
		}
		custom_description = {
			text = bridges_leyawiin_02_must_be_disabled
			NOT = { title:b_leyawiin_castle.title_province = { has_building = bridges_leyawiin_02 } }
		}
	}
}

ek_hist_tree_faith_trigger = {
	custom_tooltip = {
		text = argonian_or_hist_pantheon_or_syncretic
		OR = {
			scope:holder.culture = { has_cultural_pillar = heritage_argonian }
			scope:holder.faith = { has_doctrine = doctrine_pantheon_hist }
			scope:holder.faith = { has_doctrine_parameter = can_hold_hist_holdings }
		}
	}
}
